\def\title[#1]{
    \newpage
    \vspace*{-0.5cm}
    \ifhmode\par\fi
    \removelastskip
    \vskip 1ex\goodbreak
    \refstepcounter{chapter}%
    \noindent
    \begingroup
    \leavevmode\Large\bfseries\raggedright
    \rlap{\thechapter}%
    \hspace{5mm} \huge Title \arabic{chapter}. %
    #1
    \par
    \endgroup
    \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#1}
    %\vskip 0ex \nobreak
    %\removelastskip
    \leftskip=12mm%\relax%
}
